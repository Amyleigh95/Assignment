import matplotlib
import matplotlib.pyplot as plt
import numpy as np


labels = ['Cleveland', 'Durham', 'Northumbria']
November_2019 = [1844, 1387, 3276]
November_2020 = [2710, 1429, 6017]

x = np.arange(len(labels))
width = 0.35  

fig, ax = plt.subplots()
rects1 = ax.bar(x - width/2, November_2019, width, label='November 2019')
rects2 = ax.bar(x + width/2, November_2020, width, label='November 2020')

ax.set_ylabel('Number of cases')
ax.set_ylabel('Name of Constabulary')
ax.set_title('Comparing 2019 and 2020 Anti Social Behaviour Cases in 3 different Constabularies')
ax.set_xticks(x)
ax.set_xticklabels(labels)
ax.legend()


def autolabel(rects):
    for rect in rects:
        height = rect.get_height()
        ax.annotate('{}'.format(height),
                    xy=(rect.get_x() + rect.get_width() / 2, height),
                    xytext=(0, 3),  # 3 points vertical offset
                    textcoords="offset points",
                    ha='center', va='bottom')


autolabel(rects1)
autolabel(rects2)

fig.tight_layout()

plt.show()
